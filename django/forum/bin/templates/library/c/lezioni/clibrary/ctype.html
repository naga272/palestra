ctype



All'interno di questo header sono state dichiarate funzioni che vengono utilizzate per la classificazione dei caratteri.


Funzioni

Nome Descrizione prototype
isalnum Controlla che il carattere passato come parametro sia alfanumerico (si può passare anche un char come parametro). int isalnum(int c);
isalpha Controlla che il carattere passato come parametro sia alfabetico (si può passare anche un char come parametro). int isalpha(int c);
isblank Controlla che il carattere passato come parametro sia bianco, cioè non visibile a schermo (spazio o tabulazione) (si può passare anche un char come parametro). (introdotto dal C99) int isblank(int c);
iscntrl Controlla che il carattere passato come parametro sia di controllo (si può passare anche un char come parametro). int iscntrl(int c);
isdigit Controlla che il carattere passato come parametro sia numerico (si può passare anche un char come parametro). int isdigit(int c);
isgraph Controlla che il carattere passato come parametro sia grafico, cioè abbia un glifo ad esso associato. I caratteri di spaziatura, ad esempio, non sono considerati grafici (si può passare anche un char come parametro). int isgraph(int c);
islower Controlla che il carattere passato come parametro sia minuscolo (si può passare anche un char come parametro). int islower(int c);
isprint Controlla che il carattere passato come parametro sia stampabile (si può passare anche un char come parametro). int isprint(int c);
ispunct Controlla che il carattere passato come parametro sia di punteggiatura (si può passare anche un char come parametro). int ispunct(int c);
isspace Controlla che il carattere passato come parametro sia di spaziatura (si può passare anche un char come parametro). int isspace(int c);
isupper Controlla che il carattere passato come parametro sia maiuscolo (si può passare anche un char come parametro). int isupper(int c);
isxdigit Controlla che il carattere passato come parametro sia esadecimale, cioè sia compreso in 0-9, oppure a-f, oppure A-F (si può passare anche un char come parametro). int isxdigit(int c);
tolower Converte il carattere passato come parametro nel suo corrispondente minuscolo, se applicabile (si può passare anche un char come parametro). int tolower(int c);
toupper Converte il carattere passato come parametro nel suo corrispondente maiuscolo, se applicabile (si può passare anche un char come parametro). int toupper(int c);
isascii Controlla che il carattere passato come parametro sia compatibile con il set di caratteri ASCII (si può passare anche un char come parametro). int isascii(int c);
toascii Converte il carattere passato come parametro affinché sia compatibile con il set di caratteri ASCII (si può passare anche un char come parametro). int toascii(int c);
