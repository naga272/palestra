stdio



La libreria stdio sta per standard input / output e ha al suo interno un set di macro, procedure e funzioni per interagire con l'input e output.


Funzioni

Nome Descrizione prototype
fclose consente di chiudere un file che è stato aperto void fclose(FILE*);
fopen,
freopen,
fdopen 		consentono l'apertura di un file in lettura o in scrittura FILE *fopen(const char *file_name, const char *mode_of_operation);
FILE *freopen(const char *filename, const char *mode, FILE *stream);
FILE *fdopen(int fildes, const char *mode);
remove consente di rimuovere un file void remove(const char *filename);
rename consente di rinominare un file. void rename(const char *filename);
rewind Agisce come se fosse stata chiamata la funzione fseek(stream, 0L, SEEK_SET) sullo stream passatole; in aggiunta, cancella il suo indicatore d'errore void rewind(FILE *fp);
tmpfile Crea ed apre un file temporaneo, che viene poi cancellato e chiuso con fclose() FILE *tmpfile(void);
fgetc Restituisce un carattere da un file. int getc(FILE *stream);
gets Legge caratteri da stdin finché non incontra un carattere di nuova riga (newline) o un EOF e salva il risultato nell'argomento passatole (char *)
Funzione obsoleta e fortemente sconsigliata. 		char *gets(char *str);
fgets La stringa presa termina sempre con un carattere di nuova riga ('\n') o con il carattere nullo ('\0'). Restituisce una stringa presa in input (sia da file che da standard input). char *fgets(char *stringa, int len, FILE *fp);
fputc Scrive un carattere su un file. int fputc(int character, FILE *stream);
fputs Scrive una stringa su un file. int fputs(const char *str, FILE *stream);
ftell Restituisce un indicatore di posizione sul file che può essere passato alla funzione fseek(). long int ftell(FILE *stream);
fseek Si sposta attraverso un file int fseek(FILE *stream, long int offset, int whence);
fsetpos Imposta l'indicatore di posizionamento del file di uno stream associato al primo argomento (un FILE *) come memorizzato nel suo secondo argomento (un fpos_t *) int fsetpos(FILE *stream, const fpos_t *pos);
fread Legge da file dei dati di diverse dimensioni size_t fread(void *ptr, size_t size, size_t nmemb, FILE *stream);
fwrite Scrive su file dei dati di diverse dimensioni size_t fwrite(const void *ptr, size_t size, size_t nmemb, FILE *stream);
getc Legge e restituisce un carattere dallo stream passatole ed incrementa l'indicatore di posizionamento nel file; è possibile che venga implementato come macro con gli stessi effetti di fgetc(), tranne che potrebbe agire sullo stream più di una volta int getc(FILE *stream);
ungetc Restituisce allo stream almeno un carattere. Alla successiva chiamata di getc verrà passato il carattere restituito da ungetc int ungetc(int c, FILE *stream);
getchar Ha gli stessi effetti di getc(stdin) int getchar(void);
printf,
fprintf,
sprintf,
snprintf,
vprintf 		Utilizzato per stampare vari tipi di dato su stdout e su file size_t printf(const char*, ...);
int fprintf(FILE *fp, char *formato, ...);
int sprintf(char *str, const char *format, ...);
int snprintf (char *str, size_t size, const char *format, ...);
int vprintf(const char *format, va_list arg);
perror Scrive un messaggio di errore sul file descriptor stderr (= standard error). un carattere in uno stream di input. void perror(const char *str);
puts Scrive una stringa su stdout int puts(const char *str);

Macro / Costanti

Nome Descrizione
EOF Un numero intero negativo di tipo int usato per indicare la fine del file.
NULL viene usato per i puntatori (sono ugali a NULL se si verifica un errore)
SEEK_SET Un intero che può essere passato alla funzione fseek() per richiedere un posizionamento relativo rispetto alla posizione attuale nel file.
SEEK_END Un intero che può essere passato alla funzione fseek() per richiedere il posizionamento alla fine del file.
SEEK_CUR Un intero che può essere passato alla funzione fseek() per richiedere un posizionamento relativo rispetto alla posizione attuale nel file.

Tipi di dato

Nome Descrizione
FILE Una struttura contenente le informazioni su un file (o in genere uno stream), necessarie per eseguire su di esso le operazioni di input/output, come ad esempio:
  • la posizione attuale nello stream
  • un indicatore di fine file
  • un indicatore d'errore
  • un puntatore al buffer dello stream, se applicabile
size_t Un tipo intero che è il tipo del valore restituito dall'operatore sizeof

Variabili

Nome Descrizione
stdin Un puntatore a FILE che si riferisce allo stream di standard input, generalmente la tastiera. (ha valore 0)
stdout Un puntatore a FILE che si riferisce allo stream di standard output, generalmente un terminale (ha valore 1).
stderr Un puntatore a FILE che si riferisce allo stream di standard error, generalmente un terminale (ha valore 2).
